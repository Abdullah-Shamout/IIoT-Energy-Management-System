CREATE TABLE iot_devices (
    id SERIAL PRIMARY KEY, # the id devices as I said in the beginning the light bulb has id 1 the fan has id 2 the motor has id 3 
    device_name TEXT NOT NULL, # devices name 
    device_intensity TEXT, # device intensity just for the motor as we don't have intensity control for the fan or the light bulb 
    device_status TEXT, # device status on or off 
    device_total_consumption DOUBLE PRECISION NOT NULL, # total consumption for each device it should be calculate by script
);

CREATE TABLE sensor_readings (
    id SERIAL,
    iot_device_id INT NOT NULL,
    reading_time TIMESTAMPTZ NOT NULL,
    voltage DOUBLE PRECISION,
    current DOUBLE PRECISION,
    power DOUBLE PRECISION,
    PRIMARY KEY (reading_time, id) 
);

CREATE TABLE energy_consumption (
    id SERIAL PRIMARY KEY,
    iot_device_id INT NOT NULL,
    period_start TIMESTAMPTZ NOT NULL,
    period_end TIMESTAMPTZ NOT NULL,
    total_consumption DOUBLE PRECISION NOT NULL,
    remaining_budget DOUBLE PRECISION NOT NULL,
    consumption_status TEXT
);

CREATE TABLE llm_plans (
    id SERIAL PRIMARY KEY,
    iot_device_id INT NOT NULL,
    proposed_schedule JSON NOT NULL,
    estimated_consumption DOUBLE PRECISION,

    created_at TIMESTAMPTZ NOT NULL
);


CREATE TABLE system_configuration (
    id SERIAL PRIMARY KEY,
    data_collection_frequency INT NOT NULL,
    energy_budget DOUBLE PRECISION NOT NULL,

);
